=== 1. Fixed Sum ===

[Goal]
Create a pure, external function sum which takes a fixed size array of five unsigned integers and returns their sum.

[Code]
// SPDX-License-Identifier: MIT
pragma solidity 0.8.20;

contract Contract {
    function sum(uint[5] memory numbers) external pure returns (uint) {
        uint total = 0;
        for (uint i = 0; i < 5; i++) {
            total += numbers[i];
        }
        return total;
    }
}


=== 2. Sum Dynamic Array ===

[Goal]
Create a pure, external function sum which takes a dynamic size array of unsigned integers and returns their sum using the .length member.

[Code]
// SPDX-License-Identifier: MIT
pragma solidity 0.8.20;

contract Contract {
    function sum(uint[] calldata numbers) external pure returns (uint) {
        uint total = 0;
        for (uint i = 0; i < numbers.length; i++) {
            total += numbers[i];
        }
        return total;
    }
}


=== 3. Filter to Storage ===

[Goal]
Create a public dynamic sized array evenNumbers and a function filterEven that pushes all even numbers from an input array into storage.

[Code]
// SPDX-License-Identifier: MIT
pragma solidity 0.8.20;

contract Contract {
    uint[] public evenNumbers;

    function filterEven(uint[] calldata numbers) external {
        for (uint i = 0; i < numbers.length; i++) {
            if (numbers[i] % 2 == 0) {
                evenNumbers.push(numbers[i]);
            }
        }
    }
}


=== 4. Filter to Memory ===

[Goal]
Create a pure function filterEven that returns a new memory array containing only the even numbers from the input. The array must be sized exactly to fit the results.

[Code]
// SPDX-License-Identifier: MIT
pragma solidity 0.8.20;

contract Contract {
    function filterEven(uint[] calldata numbers) external pure returns (uint[] memory) {
        uint count = 0;
        for (uint i = 0; i < numbers.length; i++) {
            if (numbers[i] % 2 == 0) count++;
        }

        uint[] memory filtered = new uint[](count);
        uint index = 0;
        for (uint i = 0; i < numbers.length; i++) {
            if (numbers[i] % 2 == 0) {
                filtered[index] = numbers[i];
                index++;
            }
        }
        return filtered;
    }
}


=== 5. Stack Club (Add Members) ===

[Goal]
Create a dynamic array members. Define addMember to push addresses to the array and isMember to verify if an address exists via a loop.

[Code]
// SPDX-License-Identifier: MIT
pragma solidity 0.8.20;

contract StackClub {
    address[] public members;

    function addMember(address _newMember) external {
        members.push(_newMember);
    }

    function isMember(address _addr) public view returns (bool) {
        for (uint i = 0; i < members.length; i++) {
            if (members[i] == _addr) return true;
        }
        return false;
    }
}


=== 6. Stack Club (Pop and Security) ===

[Goal]
Add a constructor to include the deployer. Implement removeLastMember using pop. Apply a modifier so only existing members can add or remove members.

[Code]
// SPDX-License-Identifier: MIT
pragma solidity 0.8.20;

contract StackClub {
    address[] public members;

    constructor() {
        members.push(msg.sender);
    }

    modifier onlyMember() {
        require(isMember(msg.sender), "Not a member");
        _;
    }

    function addMember(address _newMember) external onlyMember {
        members.push(_newMember);
    }

    function removeLastMember() external onlyMember {
        members.pop();
    }

    function isMember(address _addr) public view returns (bool) {
        for (uint i = 0; i < members.length; i++) {
            if (members[i] == _addr) return true;
        }
        return false;
    }
}
